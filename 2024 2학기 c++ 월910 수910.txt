void save(const std::string& inFileName) {
	std::ifstream in{ inFileName };
	//읽기모드로 열기
	std::ofstream out{ "2024 2학기 c++ 월910 수910.txt",std::ios::app };

	out << "============================ = " << '\n';
	//시간기록
	auto now = std::chrono::system_clock::now();
	using namespace std::literals;
	now + -9h;
	out << "저장시간" << now << '\n';
	out << "============================ = " << '\n';
	out << '\n';

	in >> std::noskipws;
	while (in >> c) {
		out << c; // 읽은글자를파일에 저장한다
	}
	//한글자씩 읽어오기
	//읽은 글ㅈ를파일에 쓰기

#include<iostream>
#include<print>
#include<random>
#include"save.h"


using namespace std;

default_random_engine dre;
uniform_int_distribution uid{ 1, 9'999 };
int how(const void*, const void*);

int main() {
	//[문제] int 값 1000개를 저장할 메모리를 확보하라.
	//[1, 9'999]사이의 랜덤값으로 int 값을 채운다.
	// --qsort를 사용하여 오름차순으로 정렬한다.
	//int 값을 8자리마다 한개씩 하면 출력한다.

	int a[1000];

	for (int& num : a)
		num = uid(dre);
	
	qsort(a, 1000, 4, how );     //c 언어의 함수이지만, generic 함수이다. contiguous메모리만 정렬
										//특정 type만을 정렬하는 함수가 아니다.
										//아무 type이나 정렬할 수 있는 type과 무관한 함수
										//정렬을 어떻게 할지 외주를 맡김 이것을 디자인 패턴이라고함

	for (int num : a)
		print("{:8}", num);
	

	save("main.cpp");
}
int how(const void* p, const void* q) {
	//void*란 - 형식을 사용자가 원하는대로 정해쓰라는 의미
	//const - 메모리를 읽을 순 있는데 쓰기는 금지함이라는 의미
	int x = *(int*)p;
	int y = *(int*)q;
	
	if (x < y)
		return -1;
	else if(x > y)
		return 1;
	else
		return 0;

}

	int num;
	int* p;
	
public :
	MemoryMonster(int num) : num{ num } {
		p = new int[num];

		for (int i = 0; i < num; i++)
		{
			p[i] = i + 1;
		}
		cout << "생성자 - " << this << " = " << num << endl;
	}

	~MemoryMonster() {
		delete[] p;
		cout << "소멸자 - " << this << " = " << num << endl;
	}

	//복사 생성자
	MemoryMonster(const MemoryMonster& other) {
		num = other.num;

		p = new int[num];

		for (int i = 0; i < num; i++)
		{
			p[i] = other.p[i];
		}
		cout << "복사생성자 - " << this << " = " << num << endl;
	}
	void show() const {
		for (int i = 0; i < num; i++)
			cout << p[i] << " ";
		cout << endl;
	}
	
};

int main() {
	MemoryMonster a{ 100 };
	MemoryMonster b{ 333 };
	save("main.cpp");
}
